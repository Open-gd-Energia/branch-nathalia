-- 4) MATCH SEGURO ENTRE JSON (fattureweb) E FATURA
--    Regra conservadora:
--    - numero_uc do JSON = consumidor.uc
--    - mes_referencia do JSON = fatura.mes_referencia
--    - só atualiza quando houver exatamente 1 match
-- =========================================================

CREATE TEMP TABLE tmp_fatura_match AS
WITH json_src AS (
    SELECT
        fw.id AS fattureweb_id,
        fw.id_fatture_web,
        fw.json,
        fw.json #>> '{data,fatura_id}' AS fatura_id_externo,
        fw.json #>> '{data,unidade_consumidora,numero_uc}' AS numero_uc_json,
        to_date(fw.json #>> '{data,fatura,mes_referencia}', 'MM-YYYY') AS mes_ref_json,
        (fw.json #>> '{data,fatura,data_vencimento}')::date AS data_vencimento_json,
        fw.json #>> '{data,fatura,numero_nota}' AS numero_nota_json
    FROM public.fattureweb fw
    WHERE fw.json IS NOT NULL
      AND fw.json ? 'data'
      AND fw.json #>> '{data,fatura_id}' IS NOT NULL
      AND fw.json #>> '{data,unidade_consumidora,numero_uc}' IS NOT NULL
      AND fw.json #>> '{data,fatura,mes_referencia}' IS NOT NULL
),
candidatos AS (
    SELECT
        js.*,
        f.id AS fatura_id,
        COUNT(*) OVER (PARTITION BY js.fattureweb_id) AS qtd_matches
    FROM json_src js
    JOIN public.fatura f
      ON f.mes_referencia = js.mes_ref_json
    JOIN public.consumidor c
      ON c.id = f.consumidor_id
     AND c.uc = js.numero_uc_json
)
SELECT *
FROM candidatos
WHERE qtd_matches = 1;

    saldo_expirar_fp = COALESCE(f.saldo_expirar_fp, NULLIF((m.json #>> '{data,outros,demonstrativo_saldos_scee,saldo_expirar_fp}'), '')::numeric),
    fatura_origem = COALESCE(f.fatura_origem, NULLIF((m.json #>> '{data,fatura_origem}'), '')),
    fatura_tipo = COALESCE(f.fatura_tipo, NULLIF((m.json #>> '{data,fatura_tipo}'), '')),
    data_insercao_origem = COALESCE(f.data_insercao_origem, NULLIF((m.json #>> '{data,data_insercao}'), '')::timestamp),
    json_origem_fattureweb_id = COALESCE(f.json_origem_fattureweb_id, m.fattureweb_id)
FROM tmp_fatura_match m
WHERE f.id = m.fatura_id;

-- =========================================================

UPDATE public.consumidor c
SET
    nome = COALESCE(c.nome, NULLIF((m.json #>> '{data,unidade_consumidora,nome}'), '')),
    cpf_cnpj = COALESCE(c.cpf_cnpj, NULLIF((m.json #>> '{data,unidade_consumidora,cpf_cnpj}'), '')),
    inscricao_estadual = COALESCE(c.inscricao_estadual, NULLIF((m.json #>> '{data,unidade_consumidora,inscricao_estadual}'), '')),
    uc = COALESCE(c.uc, NULLIF((m.json #>> '{data,unidade_consumidora,numero_uc}'), '')),
    tipo_conexao = COALESCE(c.tipo_conexao, NULLIF((m.json #>> '{data,unidade_consumidora,instalacao,tipo_ligacao}'), '')),
    subgrupo = COALESCE(c.subgrupo, NULLIF((m.json #>> '{data,unidade_consumidora,instalacao,subgrupo}'), '')),
    atividade = COALESCE(c.atividade, NULLIF((m.json #>> '{data,unidade_consumidora,instalacao,atividade}'), '')),
    atividade_especifica = COALESCE(c.atividade_especifica, NULLIF((m.json #>> '{data,unidade_consumidora,instalacao,atividade_especifica}'), '')),
    amperagem = COALESCE(c.amperagem, NULLIF((m.json #>> '{data,unidade_consumidora,instalacao,amperagem}'), '')),
    categoria_tensao = COALESCE(c.categoria_tensao, NULLIF((m.json #>> '{data,unidade_consumidora,instalacao,categoria_tensao}'), '')),
    classificacao = CASE
        WHEN c.classificacao IS NOT NULL THEN c.classificacao
        WHEN upper(COALESCE(m.json #>> '{data,unidade_consumidora,instalacao,subgrupo}', '')) = 'B1'
         AND upper(COALESCE(m.json #>> '{data,unidade_consumidora,instalacao,classificacao}', '')) = 'RESIDENCIAL'
            THEN 'B1_RESIDENCIAL'
        WHEN upper(COALESCE(m.json #>> '{data,unidade_consumidora,instalacao,subgrupo}', '')) = 'B2'
         AND upper(COALESCE(m.json #>> '{data,unidade_consumidora,instalacao,classificacao}', '')) = 'RURAL'
            THEN 'B2_RURAL'
        WHEN upper(COALESCE(m.json #>> '{data,unidade_consumidora,instalacao,subgrupo}', '')) = 'B3'
         AND upper(COALESCE(m.json #>> '{data,unidade_consumidora,instalacao,classificacao}', '')) = 'COMERCIAL'
            THEN 'B3_COMERCIAL'
        ELSE c.classificacao
    END
FROM tmp_fatura_match m
JOIN public.fatura f ON f.id = m.fatura_id
WHERE c.id = f.consumidor_id;

INSERT INTO public.fatura_produto (
    fatura_id, ordem, descricao, unidade, quantidade, valor_total,
    tarifa_com_impostos, tarifa_sem_impostos, tributo_piscofins, tributo_icms,
    linha_original, componente, origem_energia, grupo_geracao, referencia_mes,
    bandeira_tarifaria, patamar, ajuste, ouc, mpt, opt, rs, muc
)

    p.ordem,
    NULLIF(p.obj ->> 'descricao', ''),
    NULLIF(p.obj ->> 'unidade', ''),
    NULLIF(p.obj ->> 'quantidade', '')::numeric,
    NULLIF(p.obj ->> 'valor_total', '')::numeric,
    NULLIF(p.obj ->> 'tarifa_com_impostos', '')::numeric,
    NULLIF(p.obj ->> 'tarifa_sem_impostos', '')::numeric,
    NULLIF(p.obj ->> 'tributo_piscofins', '')::numeric,
    NULLIF(p.obj ->> 'tributo_icms', '')::numeric,
    NULLIF(p.obj #>> '{caracteristicas,linha}', ''),
    NULLIF(p.obj #>> '{caracteristicas,componente}', ''),
    NULLIF(p.obj #>> '{caracteristicas,origem_energia}', ''),
    NULLIF(p.obj #>> '{caracteristicas,grupo_geracao}', ''),
    NULLIF(p.obj #>> '{caracteristicas,referencia_mes}', ''),
    NULLIF(p.obj #>> '{caracteristicas,bandeira_tarifaria}', ''),

    NULLIF(p.obj #>> '{caracteristicas,ouc}', '')::boolean,
    NULLIF(p.obj #>> '{caracteristicas,mpt}', '')::boolean,
    NULLIF(p.obj #>> '{caracteristicas,opt}', '')::boolean,
    NULLIF(p.obj #>> '{caracteristicas,rs}', '')::boolean,
    NULLIF(p.obj #>> '{caracteristicas,muc}', '')::boolean

SELECT
    fm.id,

    NULLIF(l.obj ->> 'descricao_completa', ''),

      SELECT 1
      FROM public.fatura_medidor_leitura fml
      WHERE fml.fatura_medidor_id = fm.id
        AND fml.ordem = l.ordem
  );

-- =========================================================
-- 15) RELATÓRIO DE VALIDAÇÃO NO FIM DA MIGRAÇÃO
-- =========================================================

DO $$
DECLARE
    v_json_total bigint;
    v_match_total bigint;
    v_sem_match bigint;
BEGIN
    SELECT COUNT(*) INTO v_json_total
    FROM public.fattureweb
    WHERE json IS NOT NULL
      AND json ? 'data'
      AND json #>> '{data,fatura_id}' IS NOT NULL;

    SELECT COUNT(*) INTO v_match_total
    FROM tmp_fatura_match;

    v_sem_match := v_json_total - v_match_total;

    RAISE NOTICE 'JSONs com fatura_id no fattureweb: %', v_json_total;
    RAISE NOTICE 'JSONs vinculados com segurança à tabela fatura: %', v_match_total;
    RAISE NOTICE 'JSONs sem match único e seguro: %', v_sem_match;
    RAISE NOTICE 'Migração concluída em modo seguro/aditivo.';
END $$;

COMMIT;
